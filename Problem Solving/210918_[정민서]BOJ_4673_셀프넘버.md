### 소스 코드

```c
#include <stdio.h>

int d(int n);

int main()
{
    int arr[10000] = {0,};
    
    for(int i=1; i<10001; i++)
    {
       if(d(i)<10001)
       {
           arr[d(i)] = 1;
       }
    }
    
    for(int i=1; i<10001; i++)
    {
        if(arr[i]==0)
        {
            printf("%d\n", i);
        }
    }
    return 0;
}

int d(int n)
{
    int sum = n;
    while(n!=0)
    {
        sum = sum + n%10;
        n = n/10;
    }
    return sum;
}
```
